            blockUser(userId, blockData) {
                const data = this.getData();
                const user = data.users.find(u => u.id === userId);
                if (user) {
                    user.status = 'blocked';
                    data.blockedUsers.push({
                        id: Date.now(),
                        userId: userId,
                        ...blockData,
                        blockedAt: new Date().toISOString()
                    });
                    this.saveData(data);
                }
            }

            unblockUser(userId) {
                const data = this.getData();
                const user = data.users.find(u => u.id === userId);
                if (user) {
                    user.status = 'active';
                    data.blockedUsers = data.blockedUsers.filter(b => b.userId !== userId);
                    this.saveData(data);
                }
            }
        }

        // Permission Management
        function togglePermission(section, type, enabled) {
            moderatorPermissions[section][type] = enabled;
            updatePermissionStatus(section, type, enabled);
        }

        function updatePermissionStatus(section, type, enabled) {
            const statusElement = document.getElementById(`${section}${type.charAt(0).toUpperCase() + type.slice(1)}Status`);
            if (statusElement) {
                statusElement.textContent = enabled ? 'Միացված է' : 'Անջատված է';
                statusElement.className = `permission-status ${enabled ? 'permission-enabled' : 'permission-disabled'}`;
            }
        }

        function loadPermissionsInModal(permissions) {
            if (permissions) {
                moderatorPermissions = { ...permissions };
                
                // Update UI for each permission
                Object.keys(moderatorPermissions).forEach(section => {
                    Object.keys(moderatorPermissions[section]).forEach(type => {
                        updatePermissionStatus(section, type, moderatorPermissions[section][type]);
                    });
                });
            } else {
                // Reset to defaults
                moderatorPermissions = {
                    bookings: { view: true, edit: true },
                    users: { view: false, edit: false },
                    services: { view: false, edit: false },
                    settings: { view: false, edit: false }
                };
                
                // Update UI
                updatePermissionStatus('bookings', 'view', true);
                updatePermissionStatus('bookings', 'edit', true);
                updatePermissionStatus('users', 'view', false);
                updatePermissionStatus('users', 'edit', false);
                updatePermissionStatus('services', 'view', false);
                updatePermissionStatus('services', 'edit', false);
                updatePermissionStatus('settings', 'view', false);
                updatePermissionStatus('settings', 'edit', false);
            }
        }

        // Delete Confirmation Modal Functions
        function showDeleteConfirmation(type, id, itemName) {
            deleteCallback = () => {
                try {
                    switch(type) {
                        case 'service':
                            dataManager.deleteService(id);
                            loadServices();
                            loadDashboardData();
                            showNotification('Ծառայությունը հաջողությամբ ջնջվեց։', 'success');
                            break;
                        case 'user':
                            dataManager.deleteUser(id);
                            loadUsers();
                            loadDashboardData();
                            showNotification('Օգտատերը հաջողությամբ ջնջվեց։', 'success');
                            break;
                        case 'moderator':
                            const data = dataManager.getData();
                            data.moderators = data.moderators.filter(m => m.id !== id);
                            dataManager.saveData(data);
                            loadModerators();
                            showNotification('Մոդերատորը ջնջվեց։', 'success');
                            break;
                        case 'backup':
                            let backups = JSON.parse(localStorage.getItem('barbershop_backups') || '[]');
                            backups = backups.filter(b => b.id !== id);
                            localStorage.setItem('barbershop_backups', JSON.stringify(backups));
                            loadBackups();
                            showNotification('Backup-ը ջնջված է։', 'success');
                            break;
                        case 'bot':
                            deleteBot();
                            break;
                    }
                } catch (error) {
                    console.error('Սխալ ջնջման ժամանակ', error);
                    showNotification('Ջնջման ժամանակ սխալ տեղի ունեցավ։', 'error');
                }
            };
            
            document.getElementById('deleteModal').classList.add('active');
        }

        function closeDeleteModal() {
            document.getElementById('deleteModal').classList.remove('active');
            deleteCallback = null;
        }

        function confirmDelete() {
            if (deleteCallback) {
                deleteCallback();
                closeDeleteModal();
            }
        }

        // Authentication
        function login() {
            const userType = document.getElementById('userType').value;
            const password = document.getElementById('password').value;

            if (!password.trim()) {
                showNotification('Խնդրում ենք մուտքագրել գաղտնաբառ', 'error');
                return;
            }

            if (userType === 'admin' && password === ADMIN_PASSWORD) {
                currentUser = { type: 'admin', name: 'Administrator' };
                showDashboard();
                showNotification('Բարի գալուստ, Ադմինիստրատոր', 'success');
            } else if (userType === 'moderator') {
                const data = dataManager.getData();
                const moderator = data.moderators.find(m => m.password === password && m.enabled);
                if (moderator) {
                    currentUser = { 
                        type: 'moderator', 
                        name: moderator.name, 
                        permissions: moderator.permissions 
                    };
                    showDashboard();
                    showNotification(`Բարի գալուստ, ${moderator.name}!`, 'success');
                    
                    // Apply permission restrictions
                    applyModeratorPermissions(moderator.permissions);
                } else {
                    showNotification('Մոդերատորի մուտքագրումները սխալ են, կամ հաշիվը անջատված է։', 'error');
                }
            } else {
                showNotification('Անվավեր մուտքագրման տվյալներ։', 'error');
            }
        }

        function applyModeratorPermissions(permissions) {
            // Hide/show navigation items based on permissions
            const navItems = {
                services: document.querySelector('[data-page="services"]').parentElement,
                users: document.querySelector('[data-page="users"]').parentElement,
                bookings: document.querySelector('[data-page="bookings"]').parentElement,
                settings: document.querySelector('[data-page="settings"]').parentElement
            };

            Object.keys(permissions).forEach(section => {
                const hasAnyPermission = permissions[section].view || permissions[section].edit;
                if (navItems[section]) {
                    navItems[section].style.display = hasAnyPermission ? 'block' : 'none';
                }
            });

            // Store permissions for later use
            currentUser.detailedPermissions = permissions;
        }

        function checkPermission(section, type) {
            if (currentUser.type === 'admin') return true;
            if (currentUser.type === 'moderator' && currentUser.detailedPermissions) {
                return currentUser.detailedPermissions[section] && currentUser.detailedPermissions[section][type];
            }
            return false;
        }

        function createBackup() {
            if (!checkPermission('settings', 'edit')) {
                showNotification('Դուք չունեք պահուստային պատճեններ ստեղծելու թույլտվություն', 'error');
                return;
            }

            try {
                const data = dataManager.getData();
                const backupDate = new Date();
                const backup = {
                    ...data,
                    backupDate: backupDate.toISOString()
                };
                
                // Get existing backups or initialize
                let backups = JSON.parse(localStorage.getItem('barbershop_backups') || '[]');
                
                // Create backup entry
                const backupEntry = {
                    id: Date.now(),
                    name: `Backup_${backupDate.toISOString().split('T')[0]}_${backupDate.getHours()}${backupDate.getMinutes()}`,
                    date: backupDate.toLocaleDateString(),
                    time: backupDate.toLocaleTimeString(),
                    size: JSON.stringify(backup).length,
                    data: backup
                };
                
                // Add to backups list
                backups.push(backupEntry);
                localStorage.setItem('barbershop_backups', JSON.stringify(backups));
                
                // Update UI
                loadBackups();
                showNotification('Backup-ը հաջողությամբ ստեղծվեց։', 'success');
            } catch (error) {
                console.error('Error creating backup:', error);
                showNotification('Սխալ առաջացավ backup-ը ստեղծելիս', 'error');
            }
        }

        function downloadBackup(backupId) {
            try {
                const backups = JSON.parse(localStorage.getItem('barbershop_backups') || '[]');
                const backup = backups.find(b => b.id === backupId);
                
                if (backup) {
                    const blob = new Blob([JSON.stringify(backup.data, null, 2)], { type: 'application/json' });
                    const url = URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = `${backup.name}.json`;
                    document.body.appendChild(a);
                    a.click();
                    document.body.removeChild(a);
                    URL.revokeObjectURL(url);
                    showNotification('Backup-ը ներբեռնվել է', 'success');
                } else {
                    showNotification('Backup-ը չի գտնվել!', 'error');
                }
            } catch (error) {
                console.error('Error downloading backup:', error);
                showNotification('Backup-ը ներբեռնելիս սխալ առաջացավ։!', 'error');
            }
        }

        function importBackup() {
            if (!checkPermission('settings', 'edit')) {
                showNotification('Դուք չունեք Backup ներմուծելու թույլտվություն', 'error');
                return;
            }

            const input = document.createElement('input');
            input.type = 'file';
            input.accept = '.json';
            input.onchange = function(e) {
                const file = e.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(e) {
                        try {
                            const backup = JSON.parse(e.target.result);
                            
                            // Validate backup structure
                            if (!backup.services || !backup.users || !backup.settings) {
                                throw new Error('Invalid backup format');
                            }
                            
                            dataManager.saveData(backup);
                            showNotification('Backup-ը հաջողությամբ ներմուծվեց։', 'success');
                            loadDashboardData();
                            
                            // Refresh current page
                            const currentPage = document.querySelector('.nav-link.active').getAttribute('data-page');
                            showPage(currentPage);
                        } catch (error) {
                            console.error('Error importing backup:', error);
                            showNotification('Անվավեր Backup ֆայլ', 'error');
                        }
                    };
                    reader.readAsText(file);
                }
            };
            input.click();
        }

        function loadBackups() {
            try {
                const backups = JSON.parse(localStorage.getItem('barbershop_backups') || '[]');
                const tbody = document.getElementById('backupsTable');
                tbody.innerHTML = '';
                
                const canEdit = checkPermission('settings', 'edit');
                
                backups.forEach(backup => {
                    const row = document.createElement('tr');
                    const actionsHtml = canEdit ? `
                        <div class="action-buttons">
                            <button class="btn btn-sm btn-primary" onclick="downloadBackup(${backup.id})">Ներբեռնել</button>
                            <button class="btn btn-sm btn-danger" onclick="showDeleteConfirmation('backup', ${backup.id}, '${backup.name}')">Ջնջել</button>
                        </div>
                    ` : `
                        <div class="action-buttons">
                            <button class="btn btn-sm btn-primary" onclick="downloadBackup(${backup.id})">Ներբեռնել</button>
                        </div>
                    `;

                    row.innerHTML = `
                        <td>${backup.name}</td>
                        <td>${backup.date}</td>
                        <td>${backup.time}</td>
                        <td>${formatFileSize(backup.size)}</td>
                        <td>${actionsHtml}</td>
                    `;
                    tbody.appendChild(row);
                });
            } catch (error) {
                console.error('Error loading backups:', error);
                showNotification('Սխալ առաջացավ Backup-ը բեռնելիս։!', 'error');
            }
        }

        function formatFileSize(bytes) {
            if (bytes < 1024) return bytes + ' B';
            else if (bytes < 1048576) return (bytes / 1024).toFixed(1) + ' KB';
            else return (bytes / 1048576).toFixed(1) + ' MB';
        }

        // Telegram Bot Functions
        async function sendTelegramMessage(chatId, message, keyboard = null) {
            try {
                const payload = {
                    chat_id: chatId,
                    text: message,
                    parse_mode: 'HTML'
                };
                
                if (keyboard) {
                    payload.reply_markup = JSON.stringify({
                        inline_keyboard: keyboard
                    });
                }
                
                const response = await fetch(`${BOT_API_URL}/sendMessage`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(payload)
                });
                
                const result = await response.json();
                if (!result.ok) {
                    throw new Error(result.description || 'Չհաջողվեց ուղարկել հաղորդագրությունը');
                }
                
                return result;
            } catch (error) {
                console.error('Telegram-ի հաղորդագրությունը ուղարկելիս սխալ առաջացավ։', error);
                logBotActivity(`Հաղորդագրությունն ուղարկելու սխալ ${error.message}`);
                return null;
            }
        }

        async function importBotAPI() {
            if (!checkPermission('settings', 'edit')) {
                showNotification('Դուք չունեք բոտի կարգավորումները կառավարելու թույլտվություն', 'error');
                return;
            }

            const token = document.getElementById('botToken').value.trim();
            
            if (!token) {
                showNotification('Խնդրում ենք մուտքագրել Bot Token', 'error');
                return;
            }
            
            // Validate token format
            if (!token.match(/^\d+:[A-Za-z0-9_-]+$/)) {
                showNotification('Անվավեր Bot Token ձևաչափ', 'error');
                return;
            }
            
            try {
                // Show loading state
                const importBtn = document.querySelector('button[onclick="importBotAPI()"]');
                const originalText = importBtn.textContent;
                importBtn.textContent = 'Միանում է...';
                importBtn.disabled = true;
                
                // Test the bot token
                const response = await fetch(`https://api.telegram.org/bot${token}/getMe`);
                const data = await response.json();
                
                if (data.ok) {
                    botInfo = data.result;
                    botConnected = true;
                    
                    // Update UI
                    document.getElementById('botUsername').value = `@${botInfo.username}`;
                    document.getElementById('botStatus').textContent = 'Միացված է';
                    document.getElementById('botStatus').className = 'status-badge status-active';
                    document.getElementById('botConfigPanel').style.display = 'block';
                    
                    // Save bot configuration
                    const botConfig = {
                        token: token,
                        username: botInfo.username,
                        firstName: botInfo.first_name,
                        connected: true,
                        connectedAt: new Date().toISOString()
                    };
                    
                    localStorage.setItem('telegram_bot_config', JSON.stringify(botConfig));
                    
                    // Initialize bot commands
                    await setupBotCommands(token);
                    
                    // Start polling for updates (simulation)
                    startBotPolling(token);
                    
                    showNotification(`Bot-ը "${botInfo.first_name}" հաջողությամբ միացված է`, 'success');
                    logBotActivity(`Բոտը միացված է ${botInfo.first_name} (@${botInfo.username})`);
                    
                } else {
                    showNotification(`Bot API Սխալ: ${data.description}`, 'error');
                    logBotActivity(`Կապը ձախողվեց ${data.description}`);
                }
                
                // Reset button state
                importBtn.textContent = originalText;
                importBtn.disabled = false;
                
            } catch (error) {
                showNotification('Failed to connect to Telegram API!', 'error');
                logBotActivity(`Միացման սխալ ${error.message}`);
                console.error('Bot-ի միացման սխալ', error);
                
                // Reset button state
                const importBtn = document.querySelector('button[onclick="importBotAPI()"]');
                importBtn.textContent = 'Ներմուծել Bot API-ն';
                importBtn.disabled = false;
            }
        }

        function logBotActivity(message) {
            try {
                const container = document.getElementById('activityLogContainer');
                const timestamp = new Date().toLocaleTimeString();
                
                // Create new log entry
                const logEntry = document.createElement('div');
                logEntry.style.marginBottom = '0.5rem';
                logEntry.style.padding = '0.5rem';
                logEntry.style.backgroundColor = 'var(--background-color)';
                logEntry.style.borderRadius = '0.25rem';
                logEntry.innerHTML = `<small style="color: var(--text-secondary);">[${timestamp}]</small> ${message}`;
                
                // Remove "No activity" message if it exists
                if (container.children.length === 1 && container.children[0].tagName === 'P') {
                    container.innerHTML = '';
                }
                
                // Add new entry at the top
                container.insertBefore(logEntry, container.firstChild);
                
                // Keep only last 10 entries
                while (container.children.length > 10) {
                    container.removeChild(container.lastChild);
                }
            } catch (error) {
                console.error('Սխալ է գրանցվել բոտի ակտիվության գրանցման ժամանակ', error);

                        } else {
                            showNotification('Webhook-ը հեռացվել է, օգտագործվում է հարցման ռեժիմը', 'success');
                            logBotActivity('Webhook-ը հեռացվել է, օգտագործվում է հարցման ռեժիմը');
                        }
                    } else {
                        showNotification(`Webhook-ի սխալ ${data.description}`, 'error');
                        logBotActivity(`Webhook-ի սխալ ${data.description}`);
                    }
                    
                    webhookBtn.textContent = originalText;
                    webhookBtn.disabled = false;
                    
                } catch (error) {
                    showNotification('Սխալ առաջացավ Webhook-ը կարգավորելիս։!', 'error');
                    logBotActivity(`Webhook-ի կարգավորման սխալ ${error.message}`);
                    
                    const webhookBtn = document.querySelector('button[onclick="setupWebhook()"]');
                    webhookBtn.textContent = 'Կարգավորեք Webhook-ը';
                    webhookBtn.disabled = false;
                }
            }
        }

        // Enhanced delete function
        async function deleteBot() {
            const botConfig = JSON.parse(localStorage.getItem('telegram_bot_config') || '{}');
            
            if (botConfig.token) {
                try {
                    // Remove webhook
                    await fetch(`https://api.telegram.org/bot${botConfig.token}/deleteWebhook`);
                    logBotActivity('Բոտը անջատված է և Webhook-ը հեռացված է');
                } catch (error) {
                    console.error('Սխալ առաջացավ  Webhook-ը հեռացնելիս', error);
                    logBotActivity(`Սխալ առաջացավ  Webhook-ը հեռացնելիս ${error.message}`);
                }
            }
            
            // Clear bot configuration
            localStorage.removeItem('telegram_bot_config');
            
            // Reset UI
            document.getElementById('botToken').value = '';
            document.getElementById('botUsername').value = '';
            document.getElementById('botStatus').textContent = 'Միացված չէ';
            document.getElementById('botStatus').className = 'status-badge status-pending';
            document.getElementById('botConfigPanel').style.display = 'none';
            document.getElementById('activityLogContainer').innerHTML = '<p style="color: var(--text-secondary);">Դեռևս ակտիվություն չկա...</p>';
            
            botConnected = false;
            botInfo = null;
            
            showNotification('Bot-ը հաջողությամբ անջատվեց։', 'success');
        }

        // Load bot configuration on page load
        function loadBotConfiguration() {
            try {
                const botConfig = JSON.parse(localStorage.getItem('telegram_bot_config') || '{}');
                
                if (botConfig.connected && botConfig.token) {
                    document.getElementById('botToken').value = botConfig.token;
                    document.getElementById('botUsername').value = `@${botConfig.username}`;
                    document.getElementById('botStatus').textContent = 'Միացված է';
                    document.getElementById('botStatus').className = 'status-badge status-active';
                    document.getElementById('botConfigPanel').style.display = 'block';
                    
                    if (botConfig.settings) {
                        document.getElementById('welcomeMessage').value = botConfig.settings.welcomeMessage || '';
                        document.getElementById('autoReply').value = botConfig.settings.autoReply || 'enabled';
                    }
                    
                    botConnected = true;
                    logBotActivity(`Bot session restored: ${botConfig.firstName} (@${botConfig.username})`);
                }
            } catch (error) {
                console.error('Սխալ առաջացավ Bot-ի կարգավորումները բեռնելիս։', error);
            }
        }

        function showNotification(message, type = 'success') {
            // Remove existing notifications
            const existingNotifications = document.querySelectorAll('.notification');
            existingNotifications.forEach(notification => {
                if (document.body.contains(notification)) {
                    document.body.removeChild(notification);
                }
            });
            
            const notification = document.createElement('div');
            notification.className = `notification ${type}`;
            notification.textContent = message;
            document.body.appendChild(notification);
            
            setTimeout(() => {
                notification.classList.add('show');
            }, 100);
            
            setTimeout(() => {
                notification.classList.remove('show');
                setTimeout(() => {
                    if (document.body.contains(notification)) {
                        document.body.removeChild(notification);
                    }
                }, 300);
            }, 5000);
        }
